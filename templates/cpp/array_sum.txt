#include <iostream>
using namespace std;

int main() {
    int numbers[] = {10, 20, 30, 40, 50};

    int size = sizeof(numbers) / sizeof(numbers[0]);
    int sum = 0;

    for (int index = 0; index < size; index++) {
        sum += numbers[index];
    }

    cout << "Sum = " << sum << endl;

    return 0;
}
