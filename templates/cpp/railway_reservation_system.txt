#include <iostream>
#include <string>
using namespace std;

struct Record {
    int train;
    string passenger;
    int seat;
};

int main() {
    Record records[100];
    int count = 0;
    int choice;
    int key;

    do {
        cout << "\n===== Railway Reservation System =====";
        cout << "\n1. Add Record";
        cout << "\n2. View All Records";
        cout << "\n3. Search Record";
        cout << "\n4. Update Record";
        cout << "\n5. Delete Record";
        cout << "\n6. Exit";
        cout << "\nPlease choose an option: ";
        cin >> choice;

        switch(choice) {

            case 1:
                if(count >= 100) {
                    cout << "Cannot add more items. Storage is full.\n";
                    break;
                }
                cout << "Enter Train Number: ";
                cin >> records[count].train;
                cout << "Enter Passenger Name: ";
                cin.ignore();
                getline(cin, records[count].passenger);
                cout << "Enter Seat Number: ";
                cin >> records[count].seat;
                count++;
                cout << "Added successfully!\n";
                break;

            case 2:
                if(count == 0) {
                    cout << "No Records Found.\n";
                }
                else {
                    cout << "\n===== Records (Train Number | Passenger Name | Seat Number) =====\n";
                    for(int i = 0; i < count; i++) {
                        cout << i + 1 << ". " << records[i].train << " | " << records[i].passenger << " | " << records[i].seat << endl;
                    }
                }
                break;

            case 3:
                cout << "Enter Train Number to Search: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].train == key) {
                            cout << "Found: " << records[i].train << " | " << records[i].passenger << " | " << records[i].seat << endl;
                            found = true;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 4:
                cout << "Enter Train Number to Update: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].train == key) {
                        cout << "Enter New Train Number: ";
                        cin >> records[i].train;
                        cout << "Enter New Passenger Name: ";
                        cin.ignore();
                        getline(cin, records[i].passenger);
                        cout << "Enter New Seat Number: ";
                        cin >> records[i].seat;
                            cout << "Updated successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 5:
                cout << "Enter Train Number to Delete: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].train == key) {
                            for(int j = i; j < count - 1; j++) {
                                records[j] = records[j + 1];
                            }
                            count--;
                            cout << "Deleted successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 6:
                cout << "Thank You for Using Railway Reservation System.\n";
                break;

            default:
                cout << "Invalid Choice.\n";
        }

    } while(choice != 6);

    return 0;
}
