#include <iostream>
using namespace std;

class Student {
    public:
        string name;
        int age;

        void displayDetails() {
            cout << "Name: " << name << endl;
            cout << "Age: " << age << endl;
        }
};

int main() {
    Student student;

    student.name = "Lazy";
    student.age = 20;

    student.displayDetails();

    return 0;
}
