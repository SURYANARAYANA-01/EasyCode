#include <iostream>
using namespace std;

int main() {
    int numbers[2][3];

    for (int row = 0; row < 2; row++) {
        for (int column = 0; column < 3; column++) {
            cout << "Enter Element [" << row << "][" << column << "]: ";

            cin >> numbers[row][column];
        }
    }

    cout << "Array Elements:" << endl;

    for (int row = 0; row < 2; row++) {
        for (int column = 0; column < 3; column++) {
            cout << numbers[row][column] << " ";
        }

        cout << endl;
    }

    return 0;
}
