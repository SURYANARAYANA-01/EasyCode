int main() {

    int n;

    cout << "Enter Array Size: ";
    cin >> n;

    int arr[n];

    cout << "Enter Elements: ";

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    reverse(arr, arr + n);

    cout << "Reversed Array: ";

    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";

    return 0;
}
