#include <iostream>
#include <string>
using namespace std;

struct Record {
    int slot;
    string vehicle;
    int hours;
};

int main() {
    Record records[100];
    int count = 0;
    int choice;
    int key;

    do {
        cout << "\n===== Parking Management System =====";
        cout << "\n1. Add Record";
        cout << "\n2. View All Records";
        cout << "\n3. Search Record";
        cout << "\n4. Update Record";
        cout << "\n5. Delete Record";
        cout << "\n6. Exit";
        cout << "\nPlease choose an option: ";
        cin >> choice;

        switch(choice) {

            case 1:
                if(count >= 100) {
                    cout << "Cannot add more items. Storage is full.\n";
                    break;
                }
                cout << "Enter Slot Number: ";
                cin >> records[count].slot;
                cout << "Enter Vehicle Number: ";
                cin.ignore();
                getline(cin, records[count].vehicle);
                cout << "Enter Hours: ";
                cin >> records[count].hours;
                count++;
                cout << "Added successfully!\n";
                break;

            case 2:
                if(count == 0) {
                    cout << "No Records Found.\n";
                }
                else {
                    cout << "\n===== Records (Slot Number | Vehicle Number | Hours) =====\n";
                    for(int i = 0; i < count; i++) {
                        cout << i + 1 << ". " << records[i].slot << " | " << records[i].vehicle << " | " << records[i].hours << endl;
                    }
                }
                break;

            case 3:
                cout << "Enter Slot Number to Search: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].slot == key) {
                            cout << "Found: " << records[i].slot << " | " << records[i].vehicle << " | " << records[i].hours << endl;
                            found = true;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 4:
                cout << "Enter Slot Number to Update: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].slot == key) {
                        cout << "Enter New Slot Number: ";
                        cin >> records[i].slot;
                        cout << "Enter New Vehicle Number: ";
                        cin.ignore();
                        getline(cin, records[i].vehicle);
                        cout << "Enter New Hours: ";
                        cin >> records[i].hours;
                            cout << "Updated successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 5:
                cout << "Enter Slot Number to Delete: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].slot == key) {
                            for(int j = i; j < count - 1; j++) {
                                records[j] = records[j + 1];
                            }
                            count--;
                            cout << "Deleted successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 6:
                cout << "Thank You for Using Parking Management System.\n";
                break;

            default:
                cout << "Invalid Choice.\n";
        }

    } while(choice != 6);

    return 0;
}
