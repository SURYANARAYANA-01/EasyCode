#include <iostream>
#include <string>
using namespace std;

struct Record {
    int id;
    string name;
    int votes;
};

int main() {
    Record records[100];
    int count = 0;
    int choice;
    int key;

    do {
        cout << "\n===== Voting System =====";
        cout << "\n1. Add Record";
        cout << "\n2. View All Records";
        cout << "\n3. Search Record";
        cout << "\n4. Update Record";
        cout << "\n5. Delete Record";
        cout << "\n6. Exit";
        cout << "\nPlease choose an option: ";
        cin >> choice;

        switch(choice) {

            case 1:
                if(count >= 100) {
                    cout << "Cannot add more items. Storage is full.\n";
                    break;
                }
                cout << "Enter Candidate ID: ";
                cin >> records[count].id;
                cout << "Enter Candidate Name: ";
                cin.ignore();
                getline(cin, records[count].name);
                cout << "Enter Votes: ";
                cin >> records[count].votes;
                count++;
                cout << "Added successfully!\n";
                break;

            case 2:
                if(count == 0) {
                    cout << "No Records Found.\n";
                }
                else {
                    cout << "\n===== Records (Candidate ID | Candidate Name | Votes) =====\n";
                    for(int i = 0; i < count; i++) {
                        cout << i + 1 << ". " << records[i].id << " | " << records[i].name << " | " << records[i].votes << endl;
                    }
                }
                break;

            case 3:
                cout << "Enter Candidate ID to Search: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].id == key) {
                            cout << "Found: " << records[i].id << " | " << records[i].name << " | " << records[i].votes << endl;
                            found = true;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 4:
                cout << "Enter Candidate ID to Update: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].id == key) {
                        cout << "Enter New Candidate ID: ";
                        cin >> records[i].id;
                        cout << "Enter New Candidate Name: ";
                        cin.ignore();
                        getline(cin, records[i].name);
                        cout << "Enter New Votes: ";
                        cin >> records[i].votes;
                            cout << "Updated successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 5:
                cout << "Enter Candidate ID to Delete: ";
                cin >> key;
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].id == key) {
                            for(int j = i; j < count - 1; j++) {
                                records[j] = records[j + 1];
                            }
                            count--;
                            cout << "Deleted successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 6:
                cout << "Thank You for Using Voting System.\n";
                break;

            default:
                cout << "Invalid Choice.\n";
        }

    } while(choice != 6);

    return 0;
}
