#include <iostream>
#include <string>
using namespace std;

struct Record {
    string site;
    string username;
    string password;
};

int main() {
    Record records[100];
    int count = 0;
    int choice;
    string key;

    do {
        cout << "\n===== Password Manager =====";
        cout << "\n1. Add Record";
        cout << "\n2. View All Records";
        cout << "\n3. Search Record";
        cout << "\n4. Update Record";
        cout << "\n5. Delete Record";
        cout << "\n6. Exit";
        cout << "\nPlease choose an option: ";
        cin >> choice;

        switch(choice) {

            case 1:
                if(count >= 100) {
                    cout << "Cannot add more items. Storage is full.\n";
                    break;
                }
                cout << "Enter Site: ";
                cin.ignore();
                getline(cin, records[count].site);
                cout << "Enter Username: ";
                cin.ignore();
                getline(cin, records[count].username);
                cout << "Enter Password: ";
                cin.ignore();
                getline(cin, records[count].password);
                count++;
                cout << "Added successfully!\n";
                break;

            case 2:
                if(count == 0) {
                    cout << "No Records Found.\n";
                }
                else {
                    cout << "\n===== Records (Site | Username | Password) =====\n";
                    for(int i = 0; i < count; i++) {
                        cout << i + 1 << ". " << records[i].site << " | " << records[i].username << " | " << records[i].password << endl;
                    }
                }
                break;

            case 3:
                cout << "Enter Site to Search: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].site == key) {
                            cout << "Found: " << records[i].site << " | " << records[i].username << " | " << records[i].password << endl;
                            found = true;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 4:
                cout << "Enter Site to Update: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].site == key) {
                        cout << "Enter New Site: ";
                        cin.ignore();
                        getline(cin, records[i].site);
                        cout << "Enter New Username: ";
                        cin.ignore();
                        getline(cin, records[i].username);
                        cout << "Enter New Password: ";
                        cin.ignore();
                        getline(cin, records[i].password);
                            cout << "Updated successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 5:
                cout << "Enter Site to Delete: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(records[i].site == key) {
                            for(int j = i; j < count - 1; j++) {
                                records[j] = records[j + 1];
                            }
                            count--;
                            cout << "Deleted successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching record was found.\n";
                }
                break;

            case 6:
                cout << "Thank You for Using Password Manager.\n";
                break;

            default:
                cout << "Invalid Choice.\n";
        }

    } while(choice != 6);

    return 0;
}
