#include <iostream>
using namespace std;

int main() {
    int numbers[] = {45, 12, 78, 34, 90};

    int size = sizeof(numbers) / sizeof(numbers[0]);
    int largestElement = numbers[0];

    for (int index = 1; index < size; index++) {
        if (numbers[index] > largestElement) {
            largestElement = numbers[index];
        }
    }

    cout << "Largest Element = " << largestElement << endl;

    return 0;
}
