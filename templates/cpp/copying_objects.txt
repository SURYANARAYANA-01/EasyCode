#include <iostream>
using namespace std;

class Student {
    public:
        string name;
        int age;
};

int main() {
    Student student1;

    student1.name = "Lazy";
    student1.age = 20;

    Student student2 = student1;

    cout << "Name: " << student2.name << endl;
    cout << "Age: " << student2.age << endl;

    return 0;
}
