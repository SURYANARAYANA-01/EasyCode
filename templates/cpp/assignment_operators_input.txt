#include <iostream>
using namespace std;

int main() {
    int number;

    cout << "Enter a number: ";
    cin >> number;

    cout << "Initial Value = " << number << endl;

    number += 5;
    cout << "number += 5 : " << number << endl;

    number -= 3;
    cout << "number -= 3 : " << number << endl;

    number *= 2;
    cout << "number *= 2 : " << number << endl;

    number /= 4;
    cout << "number /= 4 : " << number << endl;

    return 0;
}
