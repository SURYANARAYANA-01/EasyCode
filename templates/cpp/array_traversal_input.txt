#include <iostream>
#include <vector>
using namespace std;

int main() {
    int size;

    cout << "Enter Array Size: ";
    cin >> size;

    vector<int> numbers(size);

    for (int index = 0; index < size; index++) {
        cout << "Enter Element "
             << index + 1
             << ": ";

        cin >> numbers[index];
    }

    cout << "Array Elements: ";

    for (int number : numbers) {
        cout << number << " ";
    }

    return 0;
}
