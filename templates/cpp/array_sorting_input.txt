#include <iostream>
#include <algorithm>

using namespace std;

int main() {

    int n;

    cout << "Enter Array Size: ";
    cin >> n;

    int arr[n];

    cout << "Enter Elements: ";

    for (int i = 0; i < n; i++)
        cin >> arr[i];

    sort(arr, arr + n);

    cout << "Sorted Array: ";

    for (int i = 0; i < n; i++)
        cout << arr[i] << " ";

    return 0;
}
