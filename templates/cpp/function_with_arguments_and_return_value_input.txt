#include <iostream>
using namespace std;

int add(int a, int b) {

    return a + b;
}

int main() {

    int num1, num2;

    cout << "Enter First Number: ";
    cin >> num1;

    cout << "Enter Second Number: ";
    cin >> num2;

    int result = add(num1, num2);

    cout << "Sum: " << result;

    return 0;
}
