#include <iostream>
#include <string>
using namespace std;

struct Task {
    string title;
    string status;
};

int main() {
    Task tasks[100];
    int count = 0;
    int choice;
    string key;

    do {
        cout << "\n===== To-Do List Application =====";
        cout << "\n1. Add Task";
        cout << "\n2. View All Tasks";
        cout << "\n3. Search Task";
        cout << "\n4. Update Task";
        cout << "\n5. Delete Task";
        cout << "\n6. Exit";
        cout << "\nPlease choose an option: ";
        cin >> choice;

        switch(choice) {

            case 1:
                if(count >= 100) {
                    cout << "Cannot add more items. Storage is full.\n";
                    break;
                }
                cout << "Enter task title: ";
                cin.ignore();
                getline(cin, tasks[count].title);
                cout << "Is the task completed? (Yes/No): ";
                cin >> tasks[count].status;
                count++;
                cout << "Task added successfully!\n";
                break;

            case 2:
                if(count == 0) {
                    cout << "No tasks available.\n";
                }
                else {
                    cout << "\n========== Task List ==========\n";
                    for(int i = 0; i < count; i++) {
                        cout << i + 1 << ". " << tasks[i].title << " | " << tasks[i].status << endl;
                    }
                }
                break;

            case 3:
                cout << "Enter the task title to search: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(tasks[i].title == key) {
                            cout << "Task Found: " << tasks[i].title << " | " << tasks[i].status << endl;
                            found = true;
                        }
                    }
                    if(!found) cout << "No matching task was found.\n";
                }
                break;

            case 4:
                cout << "Enter the task title to update: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(tasks[i].title == key) {
                        cout << "Enter the new task title: ";
                        getline(cin, tasks[i].title);
                        cout << "Is the task completed? (Yes/No): ";
                        cin >> tasks[i].status;
                            cout << "Task updated successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching task was found.\n";
                }
                break;

            case 5:
                cout << "Enter the task title to delete: ";
                cin.ignore();
                getline(cin, key);
                {
                    bool found = false;
                    for(int i = 0; i < count; i++) {
                        if(tasks[i].title == key) {
                            for(int j = i; j < count - 1; j++) {
                                tasks[j] = tasks[j + 1];
                            }
                            count--;
                            cout << "Task deleted successfully!\n";
                            found = true;
                            break;
                        }
                    }
                    if(!found) cout << "No matching task was found.\n";
                }
                break;

            case 6:
                cout << "Thank You for Using To-Do List Application.\n";
                break;

            default:
                cout << "Invalid Choice.\n";
        }

    } while(choice != 6);

    return 0;
}
