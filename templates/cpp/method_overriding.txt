#include <iostream>
using namespace std;

class Animal {
    public:
        virtual void makeSound() {
            cout << "Animal Sound" << endl;
        }
};

class Dog : public Animal {
    public:
        void makeSound() override {
            cout << "Bark" << endl;
        }
};

int main() {
    Dog dog;

    dog.makeSound();

    return 0;
}
