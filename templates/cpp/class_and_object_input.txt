#include <iostream>
using namespace std;

class Student {
    public:
        string name;
        int age;

        void displayDetails() {
            cout << "Name: " << name << endl;
            cout << "Age: " << age << endl;
        }
};

int main() {
    Student student;

    cout << "Enter Name: ";
    cin >> student.name;

    cout << "Enter Age: ";
    cin >> student.age;

    student.displayDetails();

    return 0;
}
