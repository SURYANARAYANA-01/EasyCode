  <!-- 1. Using the for attribute (matches input id) -->
  <p>
    <label for="uname">Username:</label>
    <input type="text" id="uname" name="uname" />
  </p>

  <!-- 2. Wrapping the input inside the label -->
  <p>
    <label>
      Password:
      <input type="password" name="pass" />
    </label>
  </p>
