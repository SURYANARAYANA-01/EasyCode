records = []

while True:
    print("\n===== Password Manager =====")
    print("1. Add Record")
    print("2. View All Records")
    print("3. Search Record")
    print("4. Update Record")
    print("5. Delete Record")
    print("6. Exit")

    choice = input("Please choose an option: ")

    if choice == "1":
        if len(records) >= 100:
            print("Cannot add more items. Storage is full.")
            continue
        record = {}
        record["site"] = input("Enter Site: ")
        record["username"] = input("Enter Username: ")
        record["password"] = input("Enter Password: ")
        records.append(record)
        print("Added successfully!")

    elif choice == "2":
        if len(records) == 0:
            print("No Records Found.")
        else:
            print("\n===== Records (Site | Username | Password) =====")
            for i, record in enumerate(records):
                print(f"{i + 1}. " + str(record["site"]) + " | " + str(record["username"]) + " | " + str(record["password"]))

    elif choice == "3":
        key = input("Enter Site to Search: ")
        found = False
        for record in records:
            if record["site"] == key:
                print("Found: " + str(record["site"]) + " | " + str(record["username"]) + " | " + str(record["password"]))
                found = True
        if not found:
            print("No matching record was found.")

    elif choice == "4":
        key = input("Enter Site to Update: ")
        found = False
        for i in range(len(records)):
            if records[i]["site"] == key:
                records[i]["site"] = input("Enter New Site: ")
                records[i]["username"] = input("Enter New Username: ")
                records[i]["password"] = input("Enter New Password: ")
                print("Updated successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "5":
        key = input("Enter Site to Delete: ")
        found = False
        for i in range(len(records)):
            if records[i]["site"] == key:
                records.pop(i)
                print("Deleted successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "6":
        print("Thank You for Using Password Manager.")
        break

    else:
        print("Invalid Choice.")
