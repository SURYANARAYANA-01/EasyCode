class Student:

    def __init__(
        self,
        name,
        age
    ):
        self.name = name
        self.age = age

name = input("Enter Name: ")

age = int(input("Enter Age: "))

student1 = Student(
    name,
    age
)

student2 = student1

print("Copied Name:", student2.name)

print("Copied Age:", student2.age)
