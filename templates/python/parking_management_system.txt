records = []

while True:
    print("\n===== Parking Management System =====")
    print("1. Add Record")
    print("2. View All Records")
    print("3. Search Record")
    print("4. Update Record")
    print("5. Delete Record")
    print("6. Exit")

    choice = input("Please choose an option: ")

    if choice == "1":
        if len(records) >= 100:
            print("Cannot add more items. Storage is full.")
            continue
        record = {}
        record["slot"] = int(input("Enter Slot Number: "))
        record["vehicle"] = input("Enter Vehicle Number: ")
        record["hours"] = int(input("Enter Hours: "))
        records.append(record)
        print("Added successfully!")

    elif choice == "2":
        if len(records) == 0:
            print("No Records Found.")
        else:
            print("\n===== Records (Slot Number | Vehicle Number | Hours) =====")
            for i, record in enumerate(records):
                print(f"{i + 1}. " + str(record["slot"]) + " | " + str(record["vehicle"]) + " | " + str(record["hours"]))

    elif choice == "3":
        key = int(input("Enter Slot Number to Search: "))
        found = False
        for record in records:
            if record["slot"] == key:
                print("Found: " + str(record["slot"]) + " | " + str(record["vehicle"]) + " | " + str(record["hours"]))
                found = True
        if not found:
            print("No matching record was found.")

    elif choice == "4":
        key = int(input("Enter Slot Number to Update: "))
        found = False
        for i in range(len(records)):
            if records[i]["slot"] == key:
                records[i]["slot"] = int(input("Enter New Slot Number: "))
                records[i]["vehicle"] = input("Enter New Vehicle Number: ")
                records[i]["hours"] = int(input("Enter New Hours: "))
                print("Updated successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "5":
        key = int(input("Enter Slot Number to Delete: "))
        found = False
        for i in range(len(records)):
            if records[i]["slot"] == key:
                records.pop(i)
                print("Deleted successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "6":
        print("Thank You for Using Parking Management System.")
        break

    else:
        print("Invalid Choice.")
