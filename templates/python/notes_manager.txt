records = []

while True:
    print("\n===== Notes Manager =====")
    print("1. Add Record")
    print("2. View All Records")
    print("3. Search Record")
    print("4. Update Record")
    print("5. Delete Record")
    print("6. Exit")

    choice = input("Please choose an option: ")

    if choice == "1":
        if len(records) >= 100:
            print("Cannot add more items. Storage is full.")
            continue
        record = {}
        record["title"] = input("Enter Note Title: ")
        record["content"] = input("Enter Note Content: ")
        records.append(record)
        print("Added successfully!")

    elif choice == "2":
        if len(records) == 0:
            print("No Records Found.")
        else:
            print("\n===== Records (Note Title | Note Content) =====")
            for i, record in enumerate(records):
                print(f"{i + 1}. " + str(record["title"]) + " | " + str(record["content"]))

    elif choice == "3":
        key = input("Enter Note Title to Search: ")
        found = False
        for record in records:
            if record["title"] == key:
                print("Found: " + str(record["title"]) + " | " + str(record["content"]))
                found = True
        if not found:
            print("No matching record was found.")

    elif choice == "4":
        key = input("Enter Note Title to Update: ")
        found = False
        for i in range(len(records)):
            if records[i]["title"] == key:
                records[i]["title"] = input("Enter New Note Title: ")
                records[i]["content"] = input("Enter New Note Content: ")
                print("Updated successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "5":
        key = input("Enter Note Title to Delete: ")
        found = False
        for i in range(len(records)):
            if records[i]["title"] == key:
                records.pop(i)
                print("Deleted successfully!")
                found = True
                break
        if not found:
            print("No matching record was found.")

    elif choice == "6":
        print("Thank You for Using Notes Manager.")
        break

    else:
        print("Invalid Choice.")
