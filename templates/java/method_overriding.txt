class Animal {

    void makeSound() {
        System.out.println("Animal Sound");
    }
}

class Dog extends Animal {

    @Override
    void makeSound() {
        System.out.println("Bark");
    }
}

public class ${CLASS_NAME} {

    public static void main(String[] args) {

        Dog dog = new Dog();

        dog.makeSound();
    }
}
