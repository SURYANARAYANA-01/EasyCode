import java.util.Scanner;

public class ${CLASS_NAME} {

    public static void main(String[] args) {

        Scanner scanner = new Scanner(System.in);

        System.out.print("Enter Age: ");
        int age = scanner.nextInt();

        System.out.print("Has License (true/false): ");

        boolean hasLicense = scanner.nextBoolean();

        System.out.println("AND Result = " + (age >= 18 && hasLicense));

        System.out.println("OR Result = " + (age >= 18 || hasLicense));

        System.out.println("NOT Result = " + (!hasLicense));

        scanner.close();
    }
}
